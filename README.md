# ConfluentKafkaAvroProducer
Confluent Kafka Avro Producer Example
confluent kafka topic consume FAC \
  --bootstrap 9092 \
  --api-key "" \
  --api-secret "" \
  --schema-registry-endpoint "" \
  --schema-registry-api-key "" \
  --schema-registry-api-secret "" \
  --key-format avro \
  --value-format avro
